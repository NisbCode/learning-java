Break interrompe um programa ao chegar em determinado local

Continue ignora td aquilo q está abaixo diminuindo assim a quantidade de um código q tá consumindo memória

public class ControleFluxo6 {
    public static void main(String[] args) {
        double valorTotal = 30000;
        for (int parcela = (int) valorTotal; parcela >= 1; parcela--) {
            double valorParcela = valorTotal / parcela;
            if (valorParcela <= 1000) {
                continue;
            }
            System.out.println("A Parcela " + parcela + " R$" + valorParcela);
        }
    }
}
