Resolver paths é mesclar eles

Lembrando q o absoluto começa com / oq o leva p a raiz do sistema, n precisa de caminhos anteriores
E relativo é basicamente uma pasta solta q vc n tá escrevendo de onde veio "C:" essas coisas base

package Estudando.javacore.X_nio;

import java.nio.file.Path;
import java.nio.file.Paths;

public class ResolvendoPaths {
    public static void main(String[] args) {
        Path dir = Paths.get("home\\Naty");
        Path arquivo = Paths.get("dev\\arquivo.txt");
        Path result = dir.resolve(arquivo);
        System.out.println(result);

        Path absoluto = Paths.get("/home/Naty");
        Path relativo = Paths.get("dev");
        Path file = Paths.get("filte.txt");

        System.out.println("1: " + absoluto.resolve(relativo));
        System.out.println("2: " + absoluto.resolve(file));
        System.out.println("3: " + relativo.resolve(file));
        System.out.println("4: " + relativo.resolve(absoluto));
        System.out.println("5: " + file.resolve(absoluto));
        System.out.println("6: " + file.resolve(relativo));    //Esse n é um válido
    }

}

1: filte.txt\dev
